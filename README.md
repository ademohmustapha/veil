# VEIL is a single self-contained HTML file — no Python, no server needed

# Open directly in any browser:
open VEIL.html            # macOS
xdg-open VEIL.html        # Linux
start VEIL.html           # Windows

# Or serve locally to avoid browser file:// restrictions:
python3 -m http.server 8080
# Then open: http://localhost:8080/VEIL.html
